JS 笔记：变量、值与类型

Posted on 2017-08-25

数据类型
number, strign, boolean, null, undefined, object

变量提升
变量提升即将变量声明提升到它所在作用域的最开始的部分。

console.log(global); // undefined
var global = 'global';
console.log(global); // global
function fn () {
　　console.log(a); // undefined
　　var a = 'aaa';
　　console.log(a); // aaa
}
fn();

之所以会是以上的打印结果，是由于变量提升，实际上上面的代码是按照以下来执行的：

var global; // 变量提升，全局作用域范围内，此时只是声明，并没有赋值
console.log(global); // undefined
global = 'global'; // 此时才赋值
console.log(global); // 打印出global
function fn () {
　　var a; // 变量提升，函数作用域范围内
　　console.log(a); // undefined
　　a = 'aaa';
　　console.log(a); // aaa
}
fn();

Boolean
建议采用严格比较，可以通过 !! 将非 boolean 值转为 boolean。
布尔操作符 && 和 || 并不会转换类型。
比较操作符总是返回 boolean 类型。

var result = 1 && 2;
console.log([result, !!result]); // [2, true]

Number
数值范围：Infinity、Number.MAX_VALUE、Number.MIN_VALUE
浮点数精度：console.log(0.2 + 0.4); // 0.6000000000000001

String
隐式类型转换：console.log([1+2, '1'+2, '1'-0+2]); // [3, "12", 3]
转整数：console.log(parseInt('100abc', 2)); // 100
转浮点数：console.log(parseFloat('12.3e5xx'); // 1230000;

对象的拷贝
直接引用：let obj2 = obj1; // 改变 obj2，原对象不会变
浅拷贝：let obj2 = Object.assign({}, obj1); // 改变 obj2，原对象属性不会变，子对象会变
深拷贝：递归函数实现。 // 原对象不会变

原型链
__proto__ 构建原型链：

var a = {x: 1}, b = {y: 2}, c = {z: 3};
b.__proto__ = a;
c.__proto__ = b;

使用 Object.create 构建原型链：

var a = {x: 1};
var b = Object.create(a);
b.y = 2;
var c = Object.create(b);
c.z = 3;

使用构造器方式：（标准）

function A() { this.x = 1; }
function B() { this.y = 2; }
B.prototype = new A();
function C() { this.z = 3; }
C.prototype = new B();
var c = new C();

内置类型
Object, Function, Array, Date, Math, Regex, Error, Promise…
